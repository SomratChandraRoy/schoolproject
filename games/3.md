Memory Challenge: Numbers hide after a brief display
Level System: More numbers added as you progress (6→9→12→15 tiles)
Scoring: Points based on speed, accuracy, and level

        # Add more games here

    is_active = models.BooleanField(default=True)
    created_at = models.DateTimeField(auto_now_add=True)
    
    class Meta:
        db_table = 'games'
        ordering = ['name']
    
    def __str__(self):
        return self.name


class GameSession(models.Model):
    """Track individual game sessions"""
    STATUS_CHOICES = [
        ('in_progress', 'In Progress'),
        ('completed', 'Completed'),
        ('abandoned', 'Abandoned'),
    ]
    
    user = models.ForeignKey(User, on_delete=models.CASCADE, related_name='game_sessions')
    game = models.ForeignKey(Game, on_delete=models.CASCADE, related_name='sessions')
    status = models.CharField(max_length=20, choices=STATUS_CHOICES, default='in_progress')
    
    # Game progress
    current_level = models.IntegerField(default=1, validators=[MinValueValidator(1)])
    current_score = models.IntegerField(default=0, validators=[MinValueValidator(0)])
    highest_level_reached = models.IntegerField(default=1)
    
    # Statistics
    total_moves = models.IntegerField(default=0)
    correct_moves = models.IntegerField(default=0)
    wrong_moves = models.IntegerField(default=0)
    time_elapsed = models.IntegerField(default=0, help_text="Time in seconds")
    
    # Session metadata
    game_data = models.JSONField(default=dict, blank=True)  # Store game-specific data
    started_at = models.DateTimeField(auto_now_add=True)
    completed_at = models.DateTimeField(null=True, blank=True)
    updated_at = models.DateTimeField(auto_now=True)
    
    class Meta:
        db_table = 'game_sessions'
        ordering = ['-started_at']
        indexes = [
            models.Index(fields=['user', 'game', '-started_at']),
            models.Index(fields=['status']),
        ]
    
    def __str__(self):
        return f"{self.user.username} - {self.game.name} - Level {self.current_level}"

from rest_framework import viewsets, status
from rest_framework.decorators import action
from rest_framework.response import Response
from rest_framework.permissions import IsAuthenticated
from django.db.models import Q, Max, Avg, Sum, F
from django.utils import timezone
from datetime import timedelta

from .models import Game, GameSession, UserGameStats, Leaderboard
from .serializers import (
    GameSerializer, GameSessionSerializer, GameSessionCreateSerializer,
    GameSessionUpdateSerializer, UserGameStatsSerializer, LeaderboardSerializer
)


class GameViewSet(viewsets.ReadOnlyModelViewSet):
    """
    ViewSet for viewing games
    """
    queryset = Game.objects.filter(is_active=True)
    serializer_class = GameSerializer
    permission_classes = [IsAuthenticated]
    
    @action(detail=True, methods=['get'])
    def stats(self, request, pk=None):
        """Get user's stats for a specific game"""
        game = self.get_object()
        stats, _ = UserGameStats.objects.get_or_create(
            user=request.user,
            game=game
        )
        serializer = UserGameStatsSerializer(stats)
        return Response(serializer.data)


class GameSessionViewSet(viewsets.ModelViewSet):
    """
    ViewSet for managing game sessions
    """
    serializer_class = GameSessionSerializer
    permission_classes = [IsAuthenticated]
    
    def get_queryset(self):
        return GameSession.objects.filter(user=self.request.user)
    
    def get_serializer_class(self):
        if self.action == 'create':
            return GameSessionCreateSerializer
        elif self.action in ['update', 'partial_update']:
            return GameSessionUpdateSerializer
        return GameSessionSerializer
    
    def perform_create(self, serializer):
        serializer.save(user=self.request.user)
    
    def perform_update(self, serializer):
        instance = serializer.save()
        
        # Update user stats when session is completed
        if instance.status == 'completed':
            self.update_user_stats(instance)
    
    def update_user_stats(self, session):
        """Update user statistics after session completion"""
        stats, created = UserGameStats.objects.get_or_create(
            user=session.user,
            game=session.game
        )
        
        # Update stats
        stats.total_sessions += 1
        stats.total_score += session.current_score
        stats.highest_score = max(stats.highest_score, session.current_score)
        stats.highest_level = max(stats.highest_level, session.highest_level_reached)
        stats.total_time_played += session.time_elapsed
        
        # Calculate averages
        all_sessions = GameSession.objects.filter(
            user=session.user,
            game=session.game,
            status='completed'
        )
        
        stats.average_score = all_sessions.aggregate(Avg('current_score'))['current_score__avg'] or 0
        
        total_moves = all_sessions.aggregate(Sum('total_moves'))['total_moves__sum'] or 0
        correct_moves = all_sessions.aggregate(Sum('correct_moves'))['correct_moves__sum'] or 0
        
        if total_moves > 0:
            stats.accuracy = (correct_moves / total_moves) * 100
        
        stats.save()
    
    @action(detail=False, methods=['get'])
    def active(self, request):
        """Get active session if exists"""
        session = GameSession.objects.filter(
            user=request.user,
            status='in_progress'
        ).first()
        
        if session:
            serializer = self.get_serializer(session)
            return Response(serializer.data)
        return Response({'message': 'No active session'}, status=status.HTTP_404_NOT_FOUND)
    
    @action(detail=False, methods=['get'])
    def history(self, request):
        """Get user's game history"""
        game_id = request.query_params.get('game_id')
        limit = int(request.query_params.get('limit', 10))
        
        queryset = self.get_queryset().filter(status='completed')
        
        if game_id:
            queryset = queryset.filter(game_id=game_id)
        
        sessions = queryset[:limit]
        serializer = self.get_serializer(sessions, many=True)
        return Response(serializer.data)


class LeaderboardViewSet(viewsets.ReadOnlyModelViewSet):
    """
    ViewSet for viewing leaderboards
    """
    serializer_class = LeaderboardSerializer
    permission_classes = [IsAuthenticated]
    
    def get_queryset(self):
        game_id = self.request.query_params.get('game_id')
        period = self.request.query_params.get('period', 'all_time')
        limit = int(self.request.query_params.get('limit', 10))
        
        queryset = Leaderboard.objects.all()
        
        if game_id:
            queryset = queryset.filter(game_id=game_id)
        
        queryset = queryset.filter(period=period)
        
        return queryset[:limit]
    
    @action(detail=False, methods=['get'])
    def my_rank(self, request):
        """Get current user's rank"""
        game_id = request.query_params.get('game_id')
        period = request.query_params.get('period', 'all_time')
        
        if not game_id:
            return Response(
                {'error': 'game_id is required'},
                status=status.HTTP_400_BAD_REQUEST
            )
        
        leaderboard = Leaderboard.objects.filter(
            user=request.user,
            game_id=game_id,
            period=period
        ).first()
        
        if leaderboard:
            serializer = self.get_serializer(leaderboard)
            return Response(serializer.data)
        
        return Response({'message': 'Not ranked yet'}, status=status.HTTP_404_NOT_FOUND)

  const context = useContext(ThemeContext);
  if (!context) {
    throw new Error('useTheme must be used within ThemeProvider');
  }
  return context;
};

export const ThemeProvider = ({ children }) => {
  const [theme, setTheme] = useState(() => {
    const saved = localStorage.getItem('theme');
    return saved || 'light';
  });

  useEffect(() => {
    const root = window.document.documentElement;
    root.classList.remove('light', 'dark');
    root.classList.add(theme);
    localStorage.setItem('theme', theme);
  }, [theme]);

  const toggleTheme = () => {
    setTheme((prev) => (prev === 'light' ? 'dark' : 'light'));
  };

  return (
    <ThemeContext.Provider value={{ theme, toggleTheme }}>
      {children}
    </ThemeContext.Provider>
  );
};
2.4 Number Hunt Hook (frontend/src/hooks/useNumberHunt.js)

import { useState, useEffect, useCallback, useRef } from 'react';

const LEVEL_CONFIG = {
  1: { gridSize: 3, totalNumbers: 6, time: 3000 },
  2: { gridSize: 3, totalNumbers: 9, time: 2500 },
  3: { gridSize: 4, totalNumbers: 12, time: 2000 },
  4: { gridSize: 4, totalNumbers: 16, time: 2000 },
  5: { gridSize: 5, totalNumbers: 20, time: 1500 },
  6: { gridSize: 5, totalNumbers: 25, time: 1500 },
};

const POINTS_PER_NUMBER = 10;
const TIME_BONUS_MULTIPLIER = 0.5;

export const useNumberHunt = () => {
  const [gameState, setGameState] = useState('idle'); // idle, memorize, playing, won, lost
  const [level, setLevel] = useState(1);
  const [score, setScore] = useState(0);
  const [nextNumber, setNextNumber] = useState(1);
  const [tiles, setTiles] = useState([]);
  const [revealedTiles, setRevealedTiles] = useState([]);
  const [timeElapsed, setTimeElapsed] = useState(0);
  const [mistakes, setMistakes] = useState(0);
  const [clickedTile, setClickedTile] = useState(null);
  
  const timerRef = useRef(null);
  const startTimeRef = useRef(null);
  const levelStartTimeRef = useRef(null);

  const getLevelConfig = useCallback(() => {
    const maxLevel = Math.max(...Object.keys(LEVEL_CONFIG).map(Number));
    const currentLevel = Math.min(level, maxLevel);
    return LEVEL_CONFIG[currentLevel];
  }, [level]);

  const generateTiles = useCallback(() => {
    const config = getLevelConfig();
    const totalTiles = config.gridSize * config.gridSize;
    const numbers = Array.from({ length: config.totalNumbers }, (_, i) => i + 1);
    const emptyTiles = Array.from({ length: totalTiles - config.totalNumbers }, () => null);
    
    const allTiles = [...numbers, ...emptyTiles];
    
    // Shuffle array
    for (let i = allTiles.length - 1; i > 0; i--) {
      const j = Math.floor(Math.random() * (i + 1));
      [allTiles[i], allTiles[j]] = [allTiles[j], allTiles[i]];
    }
    
    return allTiles.map((number, index) => ({
      id: index,
      number,
      isRevealed: false,
      isCorrect: false,
      isWrong: false,
    }));
  }, [getLevelConfig]);

  const startGame = useCallback(() => {
    const newTiles = generateTiles();
    setTiles(newTiles);
    setGameState('memorize');
    setNextNumber(1);
    setRevealedTiles(newTiles.filter(t => t.number !== null).map(t => t.id));
    setMistakes(0);
    setClickedTile(null);
    levelStartTimeRef.current = Date.now();
    
    if (!startTimeRef.current) {
      startTimeRef.current = Date.now();
    }

    const config = getLevelConfig();
    setTimeout(() => {
      setRevealedTiles([]);
      setGameState('playing');
    }, config.time);
  }, [generateTiles, getLevelConfig]);

  const resetGame = useCallback(() => {
    setGameState('idle');
    setLevel(1);
    setScore(0);
    setNextNumber(1);
    setTiles([]);
    setRevealedTiles([]);
    setTimeElapsed(0);
    setMistakes(0);
    setClickedTile(null);
    startTimeRef.current = null;
    levelStartTimeRef.current = null;
    
    if (timerRef.current) {
      clearInterval(timerRef.current);
    }
  }, []);

  const calculateScore = useCallback((timeSpent) => {
    const basePoints = POINTS_PER_NUMBER;
    const timeBonus = Math.max(0, Math.floor((10000 - timeSpent) * TIME_BONUS_MULTIPLIER));
    return basePoints + timeBonus;
  }, []);

  const handleTileClick = useCallback((tileId) => {
    if (gameState !== 'playing') return;
    
    const tile = tiles.find(t => t.id === tileId);
    if (!tile || tile.number === null || tile.isCorrect) return;

    setClickedTile(tileId);
    setRevealedTiles(prev => [...prev, tileId]);

    if (tile.number === nextNumber) {
      // Correct click
      setTiles(prev =>
        prev.map(t => (t.id === tileId ? { ...t, isCorrect: true } : t))
      );
      
      const timeSpent = Date.now() - levelStartTimeRef.current;
      const points = calculateScore(timeSpent);
      setScore(prev => prev + points);
      
      const config = getLevelConfig();
      if (nextNumber === config.totalNumbers) {
        // Level completed
        setTimeout(() => {
          setLevel(prev => prev + 1);
          setGameState('won');
        }, 500);
      } else {
        setNextNumber(prev => prev + 1);
      }
      
      setTimeout(() => setClickedTile(null), 300);
    } else {
      // Wrong click
      setTiles(prev =>
        prev.map(t => (t.id === tileId ? { ...t, isWrong: true } : t))
      );
      setMistakes(prev => prev + 1);
      setScore(prev => Math.max(0, prev - 5)); // Penalty
      
      setTimeout(() => {
        setTiles(prev =>
          prev.map(t => (t.id === tileId ? { ...t, isWrong: false } : t))
        );
        setRevealedTiles(prev => prev.filter(id => id !== tileId));
        setClickedTile(null);
      }, 500);
      
      if (mistakes + 1 >= 3) {
        setTimeout(() => {
          setGameState('lost');
        }, 500);
      }
    }
  }, [gameState, tiles, nextNumber, mistakes, getLevelConfig, calculateScore]);

  const continueToNextLevel = useCallback(() => {
    startGame();
  }, [startGame]);

  // Timer effect
  useEffect(() => {
    if (gameState === 'playing' || gameState === 'memorize') {
      timerRef.current = setInterval(() => {
        if (startTimeRef.current) {
          setTimeElapsed(Math.floor((Date.now() - startTimeRef.current) / 1000));
        }
      }, 1000);
    } else {
      if (timerRef.current) {
        clearInterval(timerRef.current);
      }
    }

    return () => {
      if (timerRef.current) {
        clearInterval(timerRef.current);
      }
    };
  }, [gameState]);

  return {
    gameState,
    level,
    score,
    nextNumber,
    tiles,
    revealedTiles,
    timeElapsed,
    mistakes,
    clickedTile,
    gridSize: getLevelConfig().gridSize,
    totalNumbers: getLevelConfig().totalNumbers,
    startGame,
    resetGame,
    handleTileClick,
    continueToNextLevel,
  };
};

const GameBoard = ({ tiles, gridSize, revealedTiles, onTileClick, gameState }) => {
  const getTileClasses = (tile) => {
    const baseClasses = "aspect-square rounded-lg font-bold text-2xl transition-all duration-300 transform cursor-pointer";
    
    if (tile.number === null) {
      return cn(
        baseClasses,
        "bg-gray-100 dark:bg-gray-700 cursor-not-allowed opacity-50"
      );
    }

    const isRevealed = revealedTiles.includes(tile.id);
    
    if (tile.isCorrect) {
      return cn(
        baseClasses,
        "bg-green-500 text-white scale-110 cursor-default"
      );
    }
    
    if (tile.isWrong) {
      return cn(
        baseClasses,
        "bg-red-500 text-white scale-95 animate-shake"
      );
    }
    
    if (isRevealed || gameState === 'memorize') {
      return cn(
        baseClasses,
        "bg-blue-500 text-white hover:bg-blue-600 active:scale-95"
      );
    }
    
    return cn(
      baseClasses,
      "bg-gray-200 dark:bg-gray-600 hover:bg-gray-300 dark:hover:bg-gray-500 active:scale-95"
    );
  };

  return (
    <div className="w-full max-w-2xl mx-auto">
      <div
        className="grid gap-3 p-4 bg-white dark:bg-gray-800 rounded-xl shadow-lg"
        style={{
          gridTemplateColumns: `repeat(${gridSize}, minmax(0, 1fr))`,
        }}
      >
        {tiles.map((tile) => (
          <button
            key={tile.id}
            onClick={() => onTileClick(tile.id)}
            disabled={gameState !== 'playing' || tile.number === null || tile.isCorrect}
            className={getTileClasses(tile)}
          >
            {(revealedTiles.includes(tile.id) ||

const NumberHuntGame = ({ gameId }) => {
  const {
    gameState,
    level,
    score,
    nextNumber,
    tiles,
    revealedTiles,
    timeElapsed,
    mistakes,
    clickedTile,
    gridSize,
    totalNumbers,
    startGame,
    resetGame,
    handleTileClick,
    continueToNextLevel,
  } = useNumberHunt();

  const [sessionId, setSessionId] = useState(null);
  const [saving, setSaving] = useState(false);
  const { toast } = useToast();

  // Create session when game starts
  useEffect(() => {
    if (gameState === 'memorize' && !sessionId) {
      createSession();
    }
  }, [gameState]);

  // Update session periodically
  useEffect(() => {
    if (sessionId && (gameState === 'playing' || gameState === 'memorize')) {
      const interval = setInterval(() => {
        updateSession(false);
      }, 10000); // Update every 10 seconds

      return () => clearInterval(interval);
    }
  }, [sessionId, gameState, level, score]);

  // Save session on completion
  useEffect(() => {
    if (sessionId && (gameState === 'won' || gameState === 'lost')) {
      updateSession(true);
    }
  }, [gameState]);

  const createSession = async () => {
    try {
      const session = await gameService.createSession(gameId);
      setSessionId(session.id);
    } catch (error) {
      console.error('Failed to create session:', error);
      toast({
        title: 'Error',
        description: 'Failed to start game session',
        variant: 'destructive',
      });
    }
  };

  const updateSession = async (isComplete = false) => {
    if (!sessionId) return;

      setSaving(true);

        current_level: level,
        current_score: score,
        highest_level_reached: level,
        time_elapsed: timeElapsed,
        total_moves: nextNumber - 1,
        correct_moves: nextNumber - 1 - mistakes,
        wrong_moves: mistakes,
      };

      if (isComplete) {
        data.status = gameState === 'won' ? 'completed' : 'abandoned';
      }

      await gameService.updateSession(sessionId, data);
    } catch (error) {
      console.error('Failed to update session:', error);
    } finally {
      setSaving(false);
    }
  };

  const handleStartGame = () => {
    setSessionId(null);
    startGame();
  };

  const handleResetGame = () => {
    if (sessionId) {
      updateSession(true);

    resetGame();
    setSessionId(null);

              />
            </div>

              <Button onClick={handleResetGame} variant="outline">
                <RotateCcw className="w-4 h-4 mr-2" />
                New Game
              </Button>
            </div>
          </>
        )}

        {/* Level Complete Dialog */}
        <Dialog open={gameState === 'won'} onOpenChange={() => {}}>
