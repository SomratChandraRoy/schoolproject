from django.db import models
from django.contrib.auth.models import User
from django.core.validators import MinValueValidator

class Game(models.Model):
    """Model to store different game types"""
    GAME_TYPES = [
        ('SHIP_FIND', 'Ship Find'),
        ('MEMORY_CARDS', 'Memory Cards'),
        ('PUZZLE', 'Puzzle'),
    ]
    
    name = models.CharField(max_length=100)
    slug = models.SlugField(unique=True)
    game_type = models.CharField(max_length=20, choices=GAME_TYPES)
    description = models.TextField()
    thumbnail = models.URLField(blank=True, null=True)
    is_active = models.BooleanField(default=True)
    created_at = models.DateTimeField(auto_now_add=True)
    
    class Meta:
        db_table = 'games'
        ordering = ['name']
    
    def __str__(self):
        return self.name


class GameProgress(models.Model):
    """Track user's progress in games"""
    user = models.ForeignKey(User, on_delete=models.CASCADE, related_name='game_progress')
    game = models.ForeignKey(Game, on_delete=models.CASCADE, related_name='user_progress')
    current_level = models.IntegerField(default=1, validators=[MinValueValidator(1)])
    highest_level = models.IntegerField(default=1, validators=[MinValueValidator(1)])
    total_score = models.IntegerField(default=0)
    games_played = models.IntegerField(default=0)
    last_played = models.DateTimeField(auto_now=True)
    created_at = models.DateTimeField(auto_now_add=True)
    
    class Meta:
        db_table = 'game_progress'
        unique_together = ['user', 'game']
        ordering = ['-last_played']
    
    def __str__(self):
        return f"{self.user.username} - {self.game.name} - Level {self.current_level}"


class GameScore(models.Model):
    """Store individual game session scores"""
    user = models.ForeignKey(User, on_delete=models.CASCADE, related_name='game_scores')
    game = models.ForeignKey(Game, on_delete=models.CASCADE, related_name='scores')
    level = models.IntegerField(validators=[MinValueValidator(1)])
    score = models.IntegerField(default=0)
    
    # Ship Find specific fields
    grid_size = models.IntegerField(null=True, blank=True)
    ships_found = models.IntegerField(default=0)
    total_ships = models.IntegerField(default=0)
    clicks_used = models.IntegerField(default=0)
    time_taken = models.IntegerField(help_text="Time in seconds", null=True, blank=True)
    accuracy = models.DecimalField(max_digits=5, decimal_places=2, default=0.00)
    
    # Generic game data (JSON for flexibility)
    game_data = models.JSONField(default=dict, blank=True)
    
    completed = models.BooleanField(default=False)
    created_at = models.DateTimeField(auto_now_add=True)
    
    class Meta:
        db_table = 'game_scores'
        ordering = ['-created_at']
        indexes = [
            models.Index(fields=['-score', 'game']),
            models.Index(fields=['user', 'game', '-created_at']),
        ]
    
    def __str__(self):
        return f"{self.user.username} - {self.game.name} - {self.score}"
    
    def save(self, *args, **kwargs):
        # Calculate accuracy
        if self.clicks_used > 0:
            hits = self.ships_found if self.ships_found else 0
            self.accuracy = (hits / self.clicks_used) * 100
        super().save(*args, **kwargs)

from rest_framework import viewsets, status
from rest_framework.decorators import action
from rest_framework.response import Response
from rest_framework.permissions import IsAuthenticated
from django.db.models import Avg, Sum, Max, Count, Q
from django.utils import timezone
from datetime import timedelta, date
from .models import Game, GameProgress, GameScore, Leaderboard
from .serializers import (
    GameSerializer, GameProgressSerializer, GameScoreSerializer,
    LeaderboardSerializer, UserGameStatsSerializer

class GameViewSet(viewsets.ReadOnlyModelViewSet):
    """ViewSet for listing games"""
    queryset = Game.objects.filter(is_active=True)
    serializer_class = GameSerializer
    lookup_field = 'slug'
    
    @action(detail=True, methods=['get'])
    def details(self, request, slug=None):
        """Get detailed information about a specific game"""
        game = self.get_object()
        serializer = self.get_serializer(game)
        
        # Add additional stats if user is authenticated
        data = serializer.data
        if request.user.is_authenticated:
            try:
                progress = GameProgress.objects.get(user=request.user, game=game)
                data['user_progress'] = GameProgressSerializer(progress).data
            except GameProgress.DoesNotExist:
                data['user_progress'] = None
        
        return Response(data)


class GameProgressViewSet(viewsets.ModelViewSet):
    """ViewSet for game progress"""
    serializer_class = GameProgressSerializer
    permission_classes = [IsAuthenticated]
    
    def get_queryset(self):
        return GameProgress.objects.filter(user=self.request.user)
    
    def perform_create(self, serializer):
        serializer.save(user=self.request.user)
    
    @action(detail=False, methods=['post'])
    def update_progress(self, request):
        """Update user's game progress"""
        game_slug = request.data.get('game_slug')
        level = request.data.get('level', 1)
        score = request.data.get('score', 0)
        
        try:
            game = Game.objects.get(slug=game_slug)
        except Game.DoesNotExist:
            return Response({'error': 'Game not found'}, status=status.HTTP_404_NOT_FOUND)
        
        progress, created = GameProgress.objects.get_or_create(
            user=request.user,
            game=game,
            defaults={'current_level': level}
        )
        
        # Update progress
        progress.current_level = level
        if level > progress.highest_level:
            progress.highest_level = level
        progress.total_score += score
        progress.games_played += 1
        progress.save()
        
        return Response(GameProgressSerializer(progress).data)


class GameScoreViewSet(viewsets.ModelViewSet):
    """ViewSet for game scores"""
    serializer_class = GameScoreSerializer
    permission_classes = [IsAuthenticated]
    
    def get_queryset(self):
        queryset = GameScore.objects.filter(user=self.request.user)
        game_slug = self.request.query_params.get('game', None)
        
        if game_slug:
            queryset = queryset.filter(game__slug=game_slug)
        
        return queryset.order_by('-created_at')
    
    def perform_create(self, serializer):
        serializer.save(user=self.request.user)
        
        # Update game progress
        game = serializer.validated_data['game']
        level = serializer.validated_data['level']
        score = serializer.validated_data['score']
        
        progress, created = GameProgress.objects.get_or_create(
            user=self.request.user,
            game=game,
            defaults={'current_level': level}
        )
        
        if level > progress.highest_level:
            progress.highest_level = level
        progress.total_score += score
        progress.games_played += 1
        progress.save()
    
    @action(detail=False, methods=['get'])
    def my_stats(self, request):
        """Get user's game statistics"""
        game_slug = request.query_params.get('game')
        
        if not game_slug:
            return Response({'error': 'game slug required'}, status=status.HTTP_400_BAD_REQUEST)
        
        try:
            game = Game.objects.get(slug=game_slug)
        except Game.DoesNotExist:
            return Response({'error': 'Game not found'}, status=status.HTTP_404_NOT_FOUND)
        
        scores = GameScore.objects.filter(user=request.user, game=game)
        
        stats = scores.aggregate(
            total_games_played=Count('id'),
            total_score=Sum('score'),
            average_score=Avg('score'),
            highest_level=Max('level'),
            best_accuracy=Max('accuracy'),
            total_time_played=Sum('time_taken')
        )
        
        # Set defaults for None values
        for key, value in stats.items():
            if value is None:
                stats[key] = 0
        
        serializer = UserGameStatsSerializer(stats)
        return Response(serializer.data)
    
    @action(detail=False, methods=['get'])
    def recent_scores(self, request):
        """Get recent scores"""
        game_slug = request.query_params.get('game')
        limit = int(request.query_params.get('limit', 10))
        
        queryset = self.get_queryset()
        if game_slug:
            queryset = queryset.filter(game__slug=game_slug)
        
        scores = queryset[:limit]
        serializer = self.get_serializer(scores, many=True)
        return Response(serializer.data)


class LeaderboardViewSet(viewsets.ReadOnlyModelViewSet):
    """ViewSet for leaderboards"""
    serializer_class = LeaderboardSerializer
    
    def get_queryset(self):
        queryset = Leaderboard.objects.all()
        game_slug = self.request.query_params.get('game')
        period = self.request.query_params.get('period', 'ALL_TIME')
        
        if game_slug:
            queryset = queryset.filter(game__slug=game_slug)
        
        queryset = queryset.filter(period=period)
        
        # Get current period dates
        today = date.today()
        if period == 'DAILY':
            queryset = queryset.filter(period_start=today)
        elif period == 'WEEKLY':
            week_start = today - timedelta(days=today.weekday())
            queryset = queryset.filter(period_start=week_start)
        elif period == 'MONTHLY':
            month_start = today.replace(day=1)
            queryset = queryset.filter(period_start=month_start)
        
        return queryset.order_by('rank', '-total_score')[:100]
    
    @action(detail=False, methods=['get'])
    def top_players(self, request):
        """Get top players for a game"""
        game_slug = request.query_params.get('game')
        limit = int(request.query_params.get('limit', 10))
        
        if not game_slug:
            return Response({'error': 'game slug required'}, status=status.HTTP_400_BAD_REQUEST)
        
        try:
            game = Game.objects.get(slug=game_slug)
        except Game.DoesNotExist:
            return Response({'error': 'Game not found'}, status=status.HTTP_404_NOT_FOUND)
        
        # Get all-time top scores
        top_scores = GameScore.objects.filter(game=game)\
            .values('user__id', 'user__username')\
            .annotate(
                total_score=Sum('score'),
                games_count=Count('id'),
                best_score=Max('score'),
                average_score=Avg('score')
            )\
            .order_by('-total_score')[:limit]
        
        return Response(top_scores)

export const calculateScore = (level, clicks, ships, timeInSeconds) => {
  const baseScore = 1000;
  const levelMultiplier = level;
  const efficiencyBonus = Math.max(0, (ships * 3 - clicks) * 10);
  const timeBonus = Math.max(0, 500 - timeInSeconds);
  
  return Math.floor((baseScore + efficiencyBonus + timeBonus) * levelMultiplier);
};

export const getGridSize = (level) => {
  if (level <= 3) return 6;
  if (level <= 6) return 8;
  if (level <= 10) return 10;
  return 12;
};

export const getShipCount = (level) => {
  return Math.min(Math.floor(level / 2) + 2, 8);
};

export const formatTime = (seconds) => {
  const mins = Math.floor(seconds / 60);
  const secs = seconds % 60;
  return `${mins}:${secs.toString().padStart(2, '0')}`;
};

import { useState, useEffect, useCallback, useRef } from 'react';
import { getGridSize, getShipCount, calculateScore } from '@/lib/utils';

const SHIP_LENGTHS = [4, 3, 3, 2, 2, 2, 1, 1];

export const useGameLogic = (level, onGameComplete) => {
  const [grid, setGrid] = useState([]);
  const [revealed, setRevealed] = useState([]);
  const [ships, setShips] = useState([]);
  const [clicks, setClicks] = useState(0);
  const [hits, setHits] = useState(0);
  const [gameStarted, setGameStarted] = useState(false);
  const [gameOver, setGameOver] = useState(false);
  const [timeElapsed, setTimeElapsed] = useState(0);
  const timerRef = useRef(null);

  const gridSize = getGridSize(level);
  const shipCount = getShipCount(level);

  // Initialize game board
  const initializeGame = useCallback(() => {
    const newGrid = Array(gridSize).fill(null).map(() => Array(gridSize).fill(0));
    const newShips = [];
    const selectedShips = SHIP_LENGTHS.slice(0, shipCount);

    // Place ships randomly
    selectedShips.forEach((length, shipIndex) => {
      let placed = false;
      let attempts = 0;
      const maxAttempts = 100;

      while (!placed && attempts < maxAttempts) {
        const horizontal = Math.random() < 0.5;
        const row = Math.floor(Math.random() * gridSize);
        const col = Math.floor(Math.random() * gridSize);

        if (canPlaceShip(newGrid, row, col, length, horizontal)) {
          placeShip(newGrid, row, col, length, horizontal, shipIndex + 1);
          newShips.push({
            id: shipIndex + 1,
            length,
            positions: getShipPositions(row, col, length, horizontal),
            found: []
          });
          placed = true;
        }
        attempts++;
      }
    });

    setGrid(newGrid);
    setShips(newShips);
    setRevealed(Array(gridSize).fill(null).map(() => Array(gridSize).fill(false)));
    setClicks(0);
    setHits(0);
    setGameOver(false);
    setGameStarted(false);
    setTimeElapsed(0);
  }, [gridSize, shipCount]);

  // Check if ship can be placed
  const canPlaceShip = (grid, row, col, length, horizontal) => {
    if (horizontal) {
      if (col + length > gridSize) return false;
      for (let i = 0; i < length; i++) {

React

import React, { useEffect, useState } from 'react';
import { useNavigate } from 'react-router-dom';
import { Card, CardContent, CardHeader, CardTitle, CardDescription } from '@/components/ui/card';
import { Button } from '@/components/ui/button';
import { Badge } from '@/components/ui/badge';
import { Play, Trophy, Clock, Target } from 'lucide-react';
import gameService from '@/services/gameService';

const GamesPage = () => {
  const navigate = useNavigate();
  const [games, setGames] = useState([]);
  const [userProgress, setUserProgress] = useState({});
  const [loading, setLoading] = useState(true);

  useEffect(() => {
    loadGames();
  }, []);

  const loadGames = async () => {
    try {
      const gamesData = await gameService.getGames();
      setGames(gamesData);

      // Load user progress for each game
      const progressPromises = gamesData.map(game =>
        gameService.getGameProgress(game.slug).catch(() => null)
      );
      const progressData = await Promise.all(progressPromises);
      
      const progressMap = {};
      gamesData.forEach((game, index) => {
        if (progressData[index] && progressData[index].length > 0) {
          progressMap[game.slug] = progressData[index][0];
        }
      });
      setUserProgress(progressMap);
    } catch (error) {
      console.error('Error loading games:', error);
    } finally {
      setLoading(false);
    }
  };

  const getGameIcon = (gameType) => {
    switch (gameType) {
      case 'SHIP_FIND':
        return '🚢';
      case 'MEMORY_CARDS':
        return '🎴';
      case 'PUZZLE':
        return '🧩';
      default:
        return '🎮';
    }
  };

  if (loading) {
    return (
      <div className="min-h-screen flex items-center justify-center">
        <div className="text-center">
          <div className="animate-spin w-12 h-12 border-4 border-primary border-t-transparent rounded-full mx-auto mb-4"></div>
          <p className="text-gray-600 dark:text-gray-400">Loading games...</p>
        </div>
      </div>
    );
  }

  return (
    <div className="min-h-screen bg-gradient-to-br from-purple-50 via-white to-blue-50 dark:from-gray-900 dark:via-gray-800 dark:to-gray-900 py-12">
      <div className="container mx-auto px-4 max-w-6xl">
        <div className="text-center mb-12">
          <h1 className="text-5xl font-bold bg-gradient-to-r from-purple-600 to-blue-600 dark:from-purple-400 dark:to-blue-400 bg-clip-text text-transparent mb-4">
            Brain Games
          </h1>
          <p className="text-xl text-gray-600 dark:text-gray-400">
            Challenge yourself and improve your skills!
          </p>
        </div>

        <div className="grid md:grid-cols-2 lg:grid-cols-3 gap-6">
          {games.map((game) => {
            const progress = userProgress[game.slug];
            
            return (
              <Card 
                key={game.id} 
                className="hover:shadow-xl transition-shadow duration-300 overflow-hidden group"
              >
                <div className="h-2 bg-gradient-to-r from-purple-500 to-blue-500" />
                
                <CardHeader>
                  <div className="flex items-start justify-between">
                    <div className="text-5xl mb-3">{getGameIcon(game.game_type)}</div>
                    {progress && (
                      <Badge variant="secondary">
                        Level {progress.current_level}
                      </Badge>
                    )}
                  </div>
                  <CardTitle className="text-2xl">{game.name}</CardTitle>
                  <CardDescription className="text-base">
                    {game.description}
                  </CardDescription>
                </CardHeader>

                <CardContent className="space-y-4">
                  {progress && (
                    <div className="grid grid-cols-3 gap-2 p-3 bg-gray-50 dark:bg-gray-800 rounded-lg">
                      <div className="text-center">
                        <Trophy className="w-5 h-5 mx-auto mb-1 text-yellow-600" />
                        <p className="text-xs text-gray-600 dark:text-gray-400">Score</p>
                        <p className="font-bold">{progress.total_score}</p>
                      </div>
                      <div className="text-center">
                        <Target className="w-5 h-5 mx-auto mb-1 text-blue-600" />
                        <p className="text-xs text-gray-600 dark:text-gray-400">Best</p>
                        <p className="font-bold">Lv {progress.highest_level}</p>
                      </div>
                      <div className="text-center">
                        <Clock className="w-5 h-5 mx-auto mb-1 text-green-600" />
                        <p className="text-xs text-gray-600 dark:text-gray-400">Played</p>
                        <p className="font-bold">{progress.games_played}</p>
                      </div>
                    </div>
                  )}

                  <Button 
                    className="w-full group-hover:scale-105 transition-transform"
                    onClick={() => navigate(`/games/${game.slug}`)}
                  >
                    <Play className="w-5 h-5 mr-2" />
                    {progress ? 'Continue Playing' : 'Start Playing'}
                  </Button>
                </CardContent>
              </Card>
            );
          })}
        </div>

        {games.length === 0 && (
          <Card className="text-center py-12">
            <CardContent>
              <p className="text-xl text-gray-600 dark:text-gray-400">
                No games available at the moment.
              </p>
            </CardContent>
          </Card>
        )}
      </div>
    </div>
  );
};

export default GamesPage;
